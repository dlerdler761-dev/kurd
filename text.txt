    <script>
        // ل ڤێرێ زانیاریێن Firebase یێن خۆ دابنێ
        const firebaseConfig = {
            apiKey: "YOUR_API_KEY",
            databaseURL: "https://YOUR_PROJECT.firebaseio.com",
            projectId: "YOUR_PROJECT_ID",
            appId: "YOUR_APP_ID"
        };
        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();
        
        let userId = localStorage.getItem('poker_user_id') || "user_" + Math.floor(Math.random() * 100000);
        localStorage.setItem('poker_user_id', userId);
        let userRef = db.ref('players/' + userId);
        let balance = 0, selectedNum = null;

        // مۆزیک
        let music = document.getElementById("bgMusic");
        function toggleMusic() {
            if (music.paused) { music.play(); document.getElementById("mBtn").innerText = "🔊"; }
            else { music.pause(); document.getElementById("mBtn").innerText = "🔇"; }
        }

        function showSection(id, btn) {
            document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));
            document.querySelectorAll('.tab-btn').forEach(b => b.classList.remove('active'));
            document.getElementById(id).classList.add('active');
            btn.classList.add('active');
        }

        userRef.on('value', (snapshot) => {
            const data = snapshot.val();
            balance = data ? data.balance : 15000;
            if(!data) userRef.set({ balance: balance });
            document.getElementById('money').innerText = balance;
        });

        const grid = document.getElementById('grid');
        for(let i=1; i<=10; i++) {
            let b = document.createElement('button'); b.className = 'num-btn'; b.innerText = i;
            b.onclick = () => {
                document.querySelectorAll('.num-btn').forEach(x => x.classList.remove('selected'));
                b.classList.add('selected'); selectedNum = i;
                document.getElementById('playBtn').disabled = false;
            };
            grid.appendChild(b);
        }

        function startBattle() {
            if (balance < 500) return;
            balance -= 500; userRef.update({ balance: balance });
            let count = 0;
            let interval = setInterval(() => {
                let random = Math.floor(Math.random() * 10) + 1;
                document.getElementById('resultNum').innerText = random;
                count++;
                if(count > 15) {
                    clearInterval(interval);
                    if(random === selectedNum) {
                        balance += 5000; userRef.update({ balance: balance });
                        alert("پیرۆزە! تە ٥٠٠٠ بردن 🎉");
                    }
                }
            }, 80);
        }

        function requestDeposit() {
            let name = document.getElementById('dName').value, amount = document.getElementById('dAmount').value, tid = document.getElementById('dTransId').value;
            if(!name || !amount || !tid) return alert("تەمام بکە");
            db.ref('deposits').push({ userId, name, amount, tid, date: new Date().toLocaleString() });
            alert("داخوازا تە هاتە ناردن!");
        }

        function requestWithdraw() {
            let name = document.getElementById('wName').value, phone = document.getElementById('wPhone').value, amount = parseInt(document.getElementById('wAmount').value);
            if(amount < 25000 || amount > balance) return alert("پارێ تە بەس نینە!");
            db.ref('withdraws').push({ userId, name, phone, amount, date: new Date().toLocaleString() });
            balance -= amount; userRef.update({ balance: balance });
            alert("داخوازا تە هاتە ناردن.");
        }
